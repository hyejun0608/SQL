  FROM WORLD.COUNTRY
 ORDER BY SURFACEAREA DESC;

## 3
```
SELECT CODE, NAME, REGION, LOCALNAME, POPULATION
  FROM WORLD.COUNTRY
 WHERE NAME = LOCALNAME AND POPULATION >= 10000000
 ORDER BY POPULATION;
```
